    ? isTSX => ({
        allowNamespaces,
        isTSX,
        jsxPragma,
        jsxPragmaFrag,
        onlyRemoveTypeImports,

        allowDeclareFields: opts.allowDeclareFields,
        allowNamespaces,
        isTSX,
        jsxPragma,
        jsxPragmaFrag,

    overrides: allExtensions

            test: /\.ts$/,

            // Babel is being called`
            test: /\.tsx$/,

          },
        ],
  };
});
